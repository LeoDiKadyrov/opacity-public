# ── Phase 7 additions ─────────────────────────────────────────────────────────

import threading


def test_wal_concurrent_writes(tmp_path):
    """4 threads simultaneously call mark_processed — WAL must handle all without error."""
    import sqlite3
    from contextlib import closing
    db = str(tmp_path / "concurrent.db")
    init_db(db)  # создаёт processed_matches + WAL

    errors = []

    def _write(i):
        try:
            mark_processed(db, f"demo_{i}.dem", 100 + i, i)
        except Exception as e:
            errors.append(str(e))

    threads = [threading.Thread(target=_write, args=(i,)) for i in range(4)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()

    assert errors == [], f"WAL concurrent write errors: {errors}"
    with closing(sqlite3.connect(db)) as conn:
        count = conn.execute("SELECT COUNT(*) FROM processed_matches").fetchone()[0]
    assert count == 4, f"Expected 4 rows, got {count}"


def test_migrate_schema_idempotent(tmp_path):
    """Calling init_db() twice on the same DB must not raise."""
    db = str(tmp_path / "idempotent.db")
    init_db(db)
    init_db(db)  # second call — must be safe


def test_demo_name_in_engagements_schema(tmp_path):
    """After init_db(), engagements table has demo_name column."""
    import sqlite3
    from contextlib import closing
    db = str(tmp_path / "schema.db")
    # Create engagements without demo_name (simulate legacy DB)
    with closing(sqlite3.connect(db)) as conn:
        conn.execute("""
            CREATE TABLE engagements (
                match_id TEXT, rt_visible_to_aim_ms REAL
            )
        """)
        conn.commit()
    init_db(db)
    with closing(sqlite3.connect(db)) as conn:
        cols = {c[1] for c in conn.execute("PRAGMA table_info(engagements)").fetchall()}
    assert "demo_name" in cols, f"demo_name missing from engagements. cols={cols}"


def test_player_steamid_in_engagements_schema(tmp_path):
    """After init_db(), engagements table has player_steamid column."""
    import sqlite3
    from contextlib import closing
    db = str(tmp_path / "schema2.db")
    with closing(sqlite3.connect(db)) as conn:
        conn.execute("CREATE TABLE engagements (match_id TEXT, rt_visible_to_aim_ms REAL)")
        conn.commit()
    init_db(db)
    with closing(sqlite3.connect(db)) as conn:
        cols = {c[1] for c in conn.execute("PRAGMA table_info(engagements)").fetchall()}
    assert "player_steamid" in cols, f"player_steamid missing. cols={cols}"


def test_get_next_match_id_uses_both_tables(tmp_path):
    """get_next_match_id takes MAX across both engagements and duel_attempts."""
    import sqlite3
    from contextlib import closing
    db = str(tmp_path / "maxid.db")
    init_db(db)
    with closing(sqlite3.connect(db)) as conn:
        conn.execute("INSERT INTO engagements (match_id) VALUES ('5')")
        conn.execute("INSERT INTO duel_attempts (match_id) VALUES ('10')")
        conn.commit()
    result = get_next_match_id(db)
    assert result == 11, f"Expected 11, got {result}"


def test_processed_matches_idempotency(tmp_path):
    """Double-marking same (demo, steamid) must result in 1 row, not IntegrityError."""
    import sqlite3
    from contextlib import closing
    db = str(tmp_path / "idem.db")
    init_db(db)
    mark_processed(db, "demo.dem", 12345, 1)
    mark_processed(db, "demo.dem", 12345, 1)  # INSERT OR REPLACE — must not raise
    with closing(sqlite3.connect(db)) as conn:
        count = conn.execute("SELECT COUNT(*) FROM processed_matches").fetchone()[0]
    assert count == 1, f"Expected 1 row after double-insert, got {count}"


def test_wal_concurrent_multiprocess(tmp_path):
    """4 OS processes simultaneously write to analytics.db — WAL must handle all without error."""
    import multiprocessing
    import sqlite3
    from contextlib import closing

    db = str(tmp_path / "multiproc.db")
    init_db(db)

    def _write_row(args):
        db_path, i = args
        import sqlite3
        from contextlib import closing
        with closing(sqlite3.connect(db_path, timeout=10)) as conn:
            conn.execute("PRAGMA journal_mode=WAL")
            conn.execute("PRAGMA busy_timeout=5000")
            with conn:
                conn.execute(
                    "INSERT OR REPLACE INTO processed_matches "
                    "(demo_filename, player_steamid, match_id, processed_at) VALUES (?,?,?,?)",
                    (f"demo_{i}.dem", 1000 + i, i, "2026-01-01T00:00:00+00:00"),
                )

    ctx = multiprocessing.get_context("spawn")
    with ctx.Pool(processes=4) as pool:
        pool.map(_write_row, [(db, i) for i in range(1, 5)])

    with closing(sqlite3.connect(db)) as conn:
        count = conn.execute("SELECT COUNT(*) FROM processed_matches").fetchone()[0]
    assert count == 4, f"Expected 4 rows from 4 processes, got {count}"

# ── Phase 7 filter / assign additions ─────────────────────────────────────────

def test_filter_unprocessed_skips_processed(tmp_path):
    """filter_unprocessed excludes demos already in processed_matches."""
    import db_utils
    db = str(tmp_path / "filter.db")
    db_utils.init_db(db)
    (tmp_path / "a.dem").touch()
    (tmp_path / "b.dem").touch()
    db_utils.mark_processed(db, "a.dem", 99999, 1)

    runner = BatchRunner(db_path=db)
    paths = [tmp_path / "a.dem", tmp_path / "b.dem"]
    result = runner.filter_unprocessed(paths, player_steamid=99999)
    assert len(result) == 1
    assert result[0].name == "b.dem"


def test_filter_unprocessed_force(tmp_path):
    """force=True returns all demos regardless of processed_matches."""
    import db_utils
    db = str(tmp_path / "force.db")
    db_utils.init_db(db)
    (tmp_path / "a.dem").touch()
    db_utils.mark_processed(db, "a.dem", 99999, 1)

    runner = BatchRunner(db_path=db)
    paths = [tmp_path / "a.dem"]
    result = runner.filter_unprocessed(paths, player_steamid=99999, force=True)
    assert len(result) == 1


def test_pre_assign_sequential(tmp_path):
    """pre_assign_match_ids on empty DB assigns 1, 2, 3 for 3 demos."""
    import db_utils
    db = str(tmp_path / "assign.db")
    db_utils.init_db(db)

    runner = BatchRunner(db_path=db)
    paths = [tmp_path / f"demo{i}.dem" for i in range(3)]
    for p in paths:
        p.touch()

    assigned = runner.pre_assign_match_ids(paths)
    assert len(assigned) == 3
    match_ids = [mid for mid, _ in assigned]
    assert match_ids == [1, 2, 3], f"Expected [1,2,3], got {match_ids}"
