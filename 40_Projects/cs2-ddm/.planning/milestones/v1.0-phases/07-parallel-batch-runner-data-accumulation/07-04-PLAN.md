```python
def _run_batch_thread(demo_paths, player_steamid, n_workers, db_path, force):
    from batch_runner import BatchRunner
```

```python
# ── Section 5: Batch Analysis ─────────────────────────────────────────────────
st.header("5. Batch Analysis")
st.caption(
    "Process all .dem files in the `for_analysis/` directory in parallel. "
    "Drop demo files there, enter SteamID64, and click Run Batch."
)

FOR_ANALYSIS_DIR = Path(BATCH_INPUT_DIR)
FOR_ANALYSIS_DIR.mkdir(exist_ok=True)

dem_files = sorted(FOR_ANALYSIS_DIR.glob("*.dem"))

if dem_files:
    st.success(f"{len(dem_files)} demo(s) found in `{BATCH_INPUT_DIR}/`")
    with st.expander("Demo files"):
        for p in dem_files:
            st.write(f"  `{p.name}`")
else:
    st.info(f"No .dem files in `{BATCH_INPUT_DIR}/` — drop files there and refresh.")

# Batch controls
col_batch_sid, col_batch_workers = st.columns(2)
with col_batch_sid:
    batch_steamid_str = st.text_input(
        "Player SteamID64 (batch)",
        placeholder="76561198386265483",
        key="batch_steamid_input",
        help="17-digit SteamID64 of the player to analyze across all demos.",
    )
with col_batch_workers:
    n_workers = st.slider(
        "Worker processes",
        min_value=1,
        max_value=16,
        value=DEFAULT_BATCH_WORKERS,
        step=1,
        help=f"Parallel worker processes. Default={DEFAULT_BATCH_WORKERS} (matches i7-11800H physical cores).",
    )

force_reprocess = st.checkbox(
    "Force reprocess (ignore processed_matches)",
    value=False,
    help="Re-run analysis even for demos already in the database.",
)

# Validate SteamID
batch_steamid_valid = False
batch_player_steamid = 0
if batch_steamid_str.strip():
    try:
        batch_player_steamid = int(batch_steamid_str.strip())
        batch_steamid_valid = True
    except ValueError:
        st.error("Invalid SteamID64 — must be a plain 17-digit number.")

batch_clicked = st.button(
    "▶ Run Batch",
    use_container_width=True,
    type="primary",
    disabled=(
        not dem_files
        or not batch_steamid_valid
        or bool(st.session_state.get("batch_running"))
    ),
)


def _run_batch_thread(
    demo_paths: list,
    player_steamid: int,
    n_workers: int,
    db_path: str,
    force: bool,
) -> None:
    """Background thread: runs BatchRunner and updates st.session_state."""
    from batch_runner import BatchRunner

    total = len(demo_paths)
    st.session_state.batch_status = {
        "total": total,
        "done": 0,
        "current": "",
        "errors": [],
    }

    def _progress(done: int, total: int, current: str) -> None:
        st.session_state.batch_status["done"] = done
        st.session_state.batch_status["current"] = current

    try:
        runner = BatchRunner(db_path=db_path, n_workers=n_workers)
        results = runner.run(
            demo_paths,
            player_steamid=player_steamid,
            tickrate=64,
            force_reprocess=force,
            progress_callback=_progress,
        )
        for r in results:
            if r.get("status") == "error":
                st.session_state.batch_status["errors"].append(
                    f"{r.get('demo', '?')}: {r.get('error', 'unknown')}"
                )
        st.session_state.batch_results = results
    except Exception as exc:
        import traceback
        st.session_state.batch_status["errors"].append(
            f"BatchRunner crashed: {exc}\n{traceback.format_exc()}"
        )
    finally:
        st.session_state.batch_running = False


if batch_clicked and batch_steamid_valid and dem_files:
    st.session_state.batch_running = True
    st.session_state.batch_results = []
    t = threading.Thread(
        target=_run_batch_thread,
        args=(list(dem_files), batch_player_steamid, n_workers, DB_PATH, force_reprocess),
        daemon=True,
    )
    t.start()

# Progress polling
if st.session_state.get("batch_running"):
    status = st.session_state.get("batch_status", {})
    done = status.get("done", 0)
    total = max(status.get("total", 1), 1)
    current = status.get("current", "")
    st.progress(
        done / total,
        text=f"Processing demo {done}/{total}: {current}",
    )
    time.sleep(0.5)
    st.rerun()

# Results summary after completion
if not st.session_state.get("batch_running") and st.session_state.get("batch_results"):
    results = st.session_state.batch_results
    ok_count = sum(1 for r in results if r.get("status") == "ok")
    err_count = sum(1 for r in results if r.get("status") == "error")
    total_eng = sum(r.get("engagements", 0) for r in results)
    total_att = sum(r.get("attempts", 0) for r in results)

    st.success(
        f"Batch complete: {ok_count}/{len(results)} demos OK — "
        f"{total_eng} engagements, {total_att} duel attempts written to analytics.db"
    )

    errors = st.session_state.get("batch_status", {}).get("errors", [])
    if errors:
        st.warning(f"{err_count} demo(s) failed:")
        for err in errors:
            st.code(err)

    with st.expander("Full results"):
        res_df = pd.DataFrame(results)
        if not res_df.empty:
            st.dataframe(res_df[["demo", "status", "engagements", "attempts"]])
```
