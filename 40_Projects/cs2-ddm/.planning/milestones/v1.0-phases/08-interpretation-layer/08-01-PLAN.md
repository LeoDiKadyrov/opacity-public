```python
def assign_tier(value: float, p25: float, p50: float, p75: float, lower_is_better: bool) -> str:
    if lower_is_better:
        if value <= p25: return "Elite"
        elif value <= p50: return "Good"
        elif value <= p75: return "Average"
        else: return "Work needed"
    else:

        elif value >= p25: return "Average"
```

```python
def get_benchmark_players(db_path: str = DB_PATH) -> list[dict]:
    """Return list of {steamid, display_name, demo_count, small_sample} for all players in analytics.db."""
    with closing(sqlite3.connect(db_path)) as conn:
        df = pd.read_sql(
            "SELECT player_steamid, COUNT(DISTINCT demo_name) as demo_count FROM engagements GROUP BY player_steamid",
            conn
        )
    result = []
    for _, row in df.iterrows():
        sid = int(row["player_steamid"])
        demo_count = int(row["demo_count"])
        small_sample = demo_count < 20
        display = PLAYER_NAMES.get(sid, str(sid))
        if small_sample:
            display += " (small sample)"
        result.append({"steamid": sid, "display_name": display, "demo_count": demo_count, "small_sample": small_sample})
    return result
```

**_get_percentiles(conn, benchmark_steamid, engagement_type) -> dict[str, tuple[float,float,float]] | None:**
Internal helper — returns per-metric (p25, p50, p75) dict from the benchmark player's distribution. Returns None if demo count < 20 (triggers fallback). Use `dropna()` before `quantile()`. Compute kill_rate + hit_rate from duel_attempts table, others from engagements table.

```python
def _get_percentiles(
    conn: sqlite3.Connection,
    benchmark_steamid: int,
    engagement_type: str,
) -> Optional[dict[str, tuple[float, float, float]]]:
    # Check demo count first
    demo_df = pd.read_sql(
        "SELECT COUNT(DISTINCT demo_name) as n FROM engagements WHERE player_steamid = ? AND engagement_type = ?",
        conn, params=(int(benchmark_steamid), engagement_type)
    )
    if demo_df["n"].iloc[0] < 20:
        return None  # triggers fallback

    # RT + crosshair percentiles from engagements
    eng_df = pd.read_sql(
        "SELECT crosshair_angle_at_t0_deg, rt_visible_to_aim_ms, rt_aim_to_hit_ms, rt_visible_to_hit_ms "
        "FROM engagements WHERE player_steamid = ? AND engagement_type = ?",
        conn, params=(int(benchmark_steamid), engagement_type)
    )
    result = {}
    for col in ["crosshair_angle_at_t0_deg", "rt_visible_to_aim_ms", "rt_aim_to_hit_ms", "rt_visible_to_hit_ms"]:
        if col not in eng_df.columns:
            continue
        s = eng_df[col].dropna()
        if len(s) < 5:
            continue
        result[col] = (float(s.quantile(0.25)), float(s.quantile(0.50)), float(s.quantile(0.75)))

    # kill_rate + hit_rate from duel_attempts — per-demo rate distribution
    da_df = pd.read_sql(
        "SELECT demo_name, SUM(was_killed) as kills, COUNT(*) as attempts, "
        "SUM(bullets_hit) as bhit, SUM(bullets_fired) as bfired "
        "FROM duel_attempts WHERE player_steamid = ? AND engagement_type = ? GROUP BY demo_name",
        conn, params=(int(benchmark_steamid), engagement_type)
    )
    if len(da_df) >= 5:
        kr = (da_df["kills"] / da_df["attempts"].replace(0, float("nan")) * 100).dropna()
        if len(kr) >= 5:
            result["kill_rate"] = (float(kr.quantile(0.25)), float(kr.quantile(0.50)), float(kr.quantile(0.75)))
        hr = (da_df["bhit"] / da_df["bfired"].replace(0, float("nan")) * 100).dropna()
        if len(hr) >= 5:
            result["hit_rate"] = (float(hr.quantile(0.25)), float(hr.quantile(0.50)), float(hr.quantile(0.75)))

    return result
```

**compute_interpretation(db_path, player_steamid, benchmark_steamid, engagement_type) -> list[dict]:**
Returns one dict per metric. Schema per row:
```
{
  "metric": str,                # column name
  "label": str,                 # human display name
  "player_value": float | None, # None if player has no data
  "tier": str,                  # Elite/Good/Average/Work needed
  "benchmark_p50": float | None,
  "gap": float | None,          # player_value - benchmark_p50 (negative = better for lower-is-better)
  "drill": str,
  "caveat": str | None,         # set for RT metrics
  "small_sample": bool,         # True if fallback thresholds used
  "bottleneck_component": str | None,  # "T0→T1" or "T1→T2" for rt_visible_to_hit_ms rows
}
```

Implementation rules:
- Always cast both steamids to int() before SQL
- Use `with closing(sqlite3.connect(db_path)) as conn:` pattern
- If percentiles is None (demo_count < 20), use `_FALLBACK_THRESHOLDS[engagement_type]` and set `small_sample=True`
- Compute player's value: for RT/crosshair — mean from engagements; for kill_rate/hit_rate — global rate (sum/count) from duel_attempts
- For hold engagement_type: set rt_visible_to_aim_ms and rt_aim_to_hit_ms rows with player_value=None, tier="n/a", drill="n/a (n<20 hold moments with valid T1)"
- Survivorship bias caveat for RT metrics: "Measured on hits only — survivorship bias applies"
- RT bottleneck component: in rt_visible_to_hit_ms row, if player_steamid has both T0→T1 and T1→T2 data, compare their tier indices; set bottleneck_component to whichever is worse

**get_worst_metric(rows) -> Optional[dict]:**
Return the row with the worst tier (tier_order = {"Work needed": 3, "Average": 2, "Good": 1, "Elite": 0}). Skip rows where tier is "n/a". Return None if all rows are n/a.
