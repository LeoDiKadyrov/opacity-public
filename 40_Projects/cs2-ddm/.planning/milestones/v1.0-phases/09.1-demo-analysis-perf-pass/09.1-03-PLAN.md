```python
_SELECTIVE_WINDOW_BEFORE_TICKS: int = 384   # ~6s before each hurt
_SELECTIVE_WINDOW_AFTER_TICKS:  int = 192   # ~3s after

def _build_tick_window_union(self, anchor_ticks: list[int]) -> list[int]:
    if not anchor_ticks:
        return []
    arrs = [
        np.arange(t - _SELECTIVE_WINDOW_BEFORE_TICKS,
                  t + _SELECTIVE_WINDOW_AFTER_TICKS,
                  dtype=np.int32)
        for t in anchor_ticks
    ]
    union = np.unique(np.concatenate(arrs))
    union = union[union >= 0]
    return union.tolist()
```
